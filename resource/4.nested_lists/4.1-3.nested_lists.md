def pascal_triangle(n=3):
    from math import factorial
    result = [1]
    for m in range(1, n + 1):
        if n - m >= 0:
            x = factorial(n) / (factorial(m) * factorial(n - m))
            result.append(int(x))
    print(*result)


def pascal_triangle_str(n):
    for i in range(n):
        pascal_triangle(i)

def chunking(x, m):
    x = x.split()
    result = list()

    while x:
        y = list()
        try:
            for i in range(m):
                y.append(x[0])
                x.pop(0)
        except IndexError:
            pass
        result.append(y)
    print(result)

def sublists_of_the_list(symbols):
    symbols = symbols.split()
    n = len(symbols)
    a = list()
    result = list([[]])

    for i in range(n+1):
        for j in range(n+1):
            a.append(symbols[i:j])

    for i in a:
        if i:
            result.append(i)
    result.sort(key=lambda i: len(i))
    print(result)


sublists_of_the_list(input())
